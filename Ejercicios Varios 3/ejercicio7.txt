#include <iostream>
using namespace std;
const int N=5;
int particicon(int valor, int v[], int a, int b, int &p){

	int i=a;
	int j=b;
	while(i!=j){
		if(v[i]<valor)
			i++;
		else if (v[i]>valor)
			j--;
		else{
			swap(v,i,j);//intercambio el valor de i y j
			i++;
			j--;
		}
	}
	p=i;
}

void resuelve(int tuercas[], int tornillos[], int a, int b ){//coste: si n=1 co, si n>1 2n+2T(n/2)
	//cojo un tuerca cualquiera y pruebo con otro tornillo y voy separando en tornillos pequeños y tornillos mas grandes
	if (b > a){
		int to = tornillos[a];
		int p;
		particion(to,tuercas, a,b,p);
		int tu=tuercas[p];
		particion(tu,tornillos,a,b,p);
		resuelve(tornillos,tuercas,a,p-1);
		resuelve(tornillos,tuercas,p+1,b);

	}
	

}


int main(){//algoritmo de particion
	int tuercas[N];
	int tornillos[N];
	tuercas[0]=1;
	tuercas[1]=3;
	tuercas[2]=2;
	tuercas[3]=7;
	tuercas[4]=5;
	
	tornillos[0]=7;
	tornillos[1]=1;
	tornillos[2]=5;
	tornillos[3]=2;
	tornillos[4]=3;
	

	system("pause");
}
